function checkMeta() {
  const html = document.getElementById('check-html').value;
  if (!html.trim()) {
    alert('Paste your HTML first.');
    return;
  }

  const parser = new DOMParser();
  const doc = parser.parseFromString(html, 'text/html');
  let checks = [];

  // Title
  const title = doc.querySelector('title')?.textContent || '';
  if (title) {
    if (title.length > 60) checks.push({ status: 'warn', title: 'Title too long', detail: `${title.length} chars` });
    else if (title.length < 30) checks.push({ status: 'warn', title: 'Title too short', detail: `${title.length} chars` });
    else checks.push({ status: 'pass', title: 'Title', detail: `${title.length} chars: "${title}"` });
  } else {
    checks.push({ status: 'fail', title: 'Missing ', detail: 'No title tag found', fix: 'Add a <title> tag (30-60 chars)' });
  }

  // Description
  const desc = doc.querySelector('meta[name="description"]')?.content || '';
  if (desc) {
    if (desc.length > 160) checks.push({ status: 'warn', title: 'Description too long', detail: `${desc.length} chars` });
    else if (desc.length < 70) checks.push({ status: 'warn', title: 'Description too short', detail: `${desc.length} chars` });
    else checks.push({ status: 'pass', title: 'Description', detail: `${desc.length} chars` });
  } else {
    checks.push({ status: 'fail', title: 'Missing meta description', detail: 'No description found', fix: 'Add <meta name="description">' });
  }

  // Robots
  const robots = doc.querySelector('meta[name="robots"]')?.content || '';
  if (robots) checks.push({ status: 'pass', title: 'Robots directive', detail: robots });
  else checks.push({ status: 'warn', title: 'No robots directive', detail: 'Default behavior applies', fix: 'Add <meta name="robots" content="index, follow">' });

  // Canonical
  const canonical = doc.querySelector('link[rel="canonical"]')?.href || '';
  if (canonical) checks.push({ status: 'pass', title: 'Canonical URL', detail: canonical });
  else checks.push({ status: 'fail', title: 'Missing canonical', detail: 'No canonical URL', fix: 'Add <link rel="canonical" href="...">' });

  // OG tags
  const ogTitle = doc.querySelector('meta[property="og:title"]')?.content || '';
  const ogDesc = doc.querySelector('meta[property="og:description"]')?.content || '';
  const ogImage = doc.querySelector('meta[property="og:image"]')?.content || '';
  const ogUrl = doc.querySelector('meta[property="og:url"]')?.content || '';

  if (ogTitle && ogDesc) checks.push({ status: 'pass', title: 'Open Graph tags', detail: 'Title and description present' });
  else checks.push({ status: 'fail', title: 'Missing Open Graph', detail: 'OG title/description not found', fix: 'Add og:title and og:description' });

  if (ogImage) checks.push({ status: 'pass', title: 'OG Image', detail: ogImage });
  else checks.push({ status: 'fail', title: 'Missing OG Image', detail: 'No image for social sharing', fix: 'Add <meta property="og:image" content="...">' });

  // Twitter
  const twCard = doc.querySelector('meta[name="twitter:card"]')?.content || '';
  const twTitle = doc.querySelector('meta[name="twitter:title"]')?.content || '';
  if (twCard) checks.push({ status: 'pass', title: 'Twitter Card', detail: twCard });
  else checks.push({ status: 'warn', title: 'No Twitter Card', detail: 'Twitter tags not found', fix: 'Add Twitter Card meta tags' });

  // Structured Data
  const jsonLdScripts = doc.querySelectorAll('script[type="application/ld+json"]');
  if (jsonLdScripts.length > 0) {
    checks.push({ status: 'pass', title: 'Structured data', detail: `${jsonLdScripts.length} JSON-LD block(s) found` });
  } else {
    checks.push({ status: 'warn', title: 'No structured data', detail: 'No JSON-LD found', fix: 'Add structured data for rich search results' });
  }

  // H1
  const h1s = doc.querySelectorAll('h1');
  if (h1s.length === 0) checks.push({ status: 'fail', title: 'No H1 tag', detail: 'Page has no heading', fix: 'Add exactly one <h1> per page' });
  else if (h1s.length > 1) checks.push({ status: 'warn', title: 'Multiple H1 tags', detail: `${h1s.length} found`, fix: 'Use only one H1 per page' });
  else checks.push({ status: 'pass', title: 'H1 tag', detail: 'One H1 found — correct' });

  // Lang
  const lang = doc.documentElement.lang || '';
  if (lang) checks.push({ status: 'pass', title: 'Language attribute', detail: lang });
  else checks.push({ status: 'warn', title: 'No lang attribute', detail: 'html tag missing lang', fix: 'Add <html lang="en">' });

  // Score
  const passCount = checks.filter(c => c.status === 'pass').length;
  const score = Math.round((passCount / checks.length) * 100);

  renderResults(score, checks, null);
}

function renderResults(score, checks, metaOutput) {
  const section = document.getElementById('results');
  section.classList.add('show');

  // Score
  const scoreSection = document.getElementById('score-section');
  let color, label;
  if (score >= 80) { color = 'var(--green)'; label = 'Excellent'; }
  else if (score >= 60) { color = 'var(--orange)'; label = 'Needs work'; }
  else { color = 'var(--red)'; label = 'Poor'; }
  scoreSection.innerHTML = `
    <div class="score-circle" style="border-color:${color}; color:${color}">${score}</div>
    <div class="score-label">${label} — ${checks.filter(c=>c.status==='pass').length} passed, ${checks.filter(c=>c.status==='warn').length} warnings, ${checks.filter(c=>c.status==='fail').length} issues</div>
  `;

  // Checks
  const checksList = document.getElementById('checks-list');
  checksList.innerHTML = checks.map(c => {
    const icon = c.status === 'pass' ? '✓' : c.status === 'warn' ? '⚠' : '✗';
    return `<div class="check check-${c.status}">
      <div class="check-icon">${icon}</div>
      <div>
        <div class="check-title">${c.title}</div>
        <div class="check-detail">${c.detail}</div>
        ${c.fix ? `<div class="check-fix">Fix: ${c.fix}</div>` : ''}
      </div>
    </div>`;
  }).join('');

  // Output
  const outputBlock = document.getElementById('output');
  const outputPre = document.getElementById('meta-output');
  if (metaOutput) {
    outputBlock.style.display = 'block';
    outputPre.textContent = metaOutput;
  } else {
    outputBlock.style.display = 'none';
  }

  section.scrollIntoView({ behavior: 'smooth' });
}

function copyOutput() {
  const text = document.getElementById('meta-output').textContent;
  navigator.clipboard.writeText(text).then(() => {
    const btn = document.querySelector('.copy-btn');
    btn.textContent = 'Copied!';
    setTimeout(() => btn.textContent = 'Copy', 2000);
  });
}
